# deploy.sh
```
#!/bin/bash

set -e

# =========================
# Config
# =========================

APP_NAME="swyp-spring"

IMAGE="app5swyp6team/swyp-spring:latest"

NETWORK="app-tier"

PROFILE_FILE="/home/ubuntu/active-profile"

NGINX_CONTAINER="nginx"

NGINX_CONFIG="/home/ubuntu/app/volumes/nginx_data/service-url.inc"

HEALTH_CHECK_URI="/actuator/health"

BLUE_PORT=8081
GREEN_PORT=8082

# =========================
# Current Profile
# =========================

if [ ! -f "$PROFILE_FILE" ]; then
    CURRENT=""
else
    CURRENT=$(cat "$PROFILE_FILE")
fi

echo "Current Profile : ${CURRENT}"

# =========================
# Determine Target
# =========================

if [ "$CURRENT" = "blue" ]; then

    TARGET="green"
    TARGET_PORT=$GREEN_PORT

    OLD="blue"

elif [ "$CURRENT" = "green" ]; then

    TARGET="blue"
    TARGET_PORT=$BLUE_PORT

    OLD="green"

else

    TARGET="blue"
    TARGET_PORT=$BLUE_PORT

    OLD=""

fi

TARGET_CONTAINER="${APP_NAME}-${TARGET}"

echo "Target Profile : ${TARGET}"

# =========================
# Pull Latest Image
# =========================

docker pull ${IMAGE}

# =========================
# Remove Existing Target
# =========================

docker rm -f ${TARGET_CONTAINER} 2>/dev/null || true

# =========================
# Run New Container
# =========================

docker run -d \
  --name ${TARGET_CONTAINER} \
  --restart unless-stopped \
  --network ${NETWORK} \
  -p ${TARGET_PORT}:8080 \
  -e SPRING_PROFILES_ACTIVE=prod \
  ${IMAGE}

# =========================
# Health Check
# =========================

echo "Health Checking..."

for i in {1..30}
do

  STATUS=$(curl -s \
      http://localhost:${TARGET_PORT}${HEALTH_CHECK_URI} \
      | grep UP || true)

  if [ -n "$STATUS" ]; then
      echo "Health Check Success"
      break
  fi

  if [ "$i" -eq 30 ]; then
      echo "Health Check Failed"

      docker logs ${TARGET_CONTAINER}

      docker rm -f ${TARGET_CONTAINER}

      exit 1
  fi

  sleep 5

done

# =========================
# Switch Nginx
# =========================

echo "server ${TARGET_CONTAINER}:8080;" | sudo tee ${NGINX_CONFIG}

cat ${NGINX_CONFIG}

docker exec ${NGINX_CONTAINER} nginx -t

docker exec ${NGINX_CONTAINER} nginx -s reload

echo "${TARGET}" > ${PROFILE_FILE}

echo "Traffic Switched To ${TARGET}"

# =========================
# Remove Old Container
# =========================

if [ -n "$OLD" ]; then

    echo "Waiting before removing old container..."
    sleep 10

    OLD_CONTAINER="${APP_NAME}-${OLD}"

    docker rm -f ${OLD_CONTAINER} || true

fi

echo "Deployment Success"
```

# 커멘트
현재 설정에서 핵심 문제 하나를 먼저 짚어야 할 것 같습니다. docker stats에 찍힌 LIMIT이 1.861GiB인데, 이게 호스트 전체 메모리(1.9Gi)와 거의 같습니다. 이건 docker run 시점에 `--memory` 옵션을 따로 안 줬다는 뜻이에요. deploy.sh를 보면 실제로 `-m` 플래그가 없죠.

이게 왜 문제가 되냐면, `-XX:+UseContainerSupport`는 cgroup에 설정된 메모리 limit을 기준으로 퍼센트를 계산하는데, limit을 따로 안 주면 JVM이 "내가 쓸 수 있는 메모리는 1.861GiB구나"라고 인식합니다. 그 상태에서 `MaxRAMPercentage=70.0`이면 힙을 최대 1.3GB 가까이 잡을 수 있다고 판단하게 되고, `InitialRAMPercentage=70.0`까지 같이 걸려 있으니 시작 시점부터 그 70%를 즉시 commit해버립니다. 빈 코드인데도 708MiB나 먹는 이유가 거의 이거라고 보시면 됩니다. 실제로 쓰는 게 아니라 시작할 때부터 미리 크게 잡아놓는 거예요.

그래서 가장 먼저 손볼 곳은 deploy.sh입니다. 컨테이너에 실제 메모리 상한을 명시적으로 줘야 JVM이 제대로 된 기준으로 퍼센트를 계산합니다.

```bash
docker run -d \
  --name ${TARGET_CONTAINER} \
  --restart unless-stopped \
  --network ${NETWORK} \
  -p ${TARGET_PORT}:8080 \
  -m 512m \
  --memory-swap 512m \
  -e SPRING_PROFILES_ACTIVE=prod \
  ${IMAGE}
```

`--memory-swap`을 같이 줘서 컨테이너가 swap까지 추가로 빌려쓰지 못하게 막는 게 좋습니다(안 그러면 컨테이너 단위로도 swap에 의존하면서 OOM 판단이 더 복잡해집니다).

다음으로 JAVA_OPTS 쪽인데, 컨테이너 limit을 512m로 명시한다고 가정하면 퍼센트 기반 대신 고정값으로 직접 박는 게 이 환경에서는 더 예측 가능합니다. 퍼센트 방식은 limit이 흔들릴 때 유연하다는 장점이 있지만, 지금처럼 인스턴스 자체가 작고 여러 컨테이너가 나눠 써야 하는 상황에서는 정확한 상한을 보장하는 게 더 중요합니다.

```dockerfile
ENV JAVA_OPTS="-Xms128m -Xmx320m \
           -XX:MetaspaceSize=64M \
           -XX:MaxMetaspaceSize=128M \
           -XX:ReservedCodeCacheSize=48m \
           -XX:+UseSerialGC \
           -XX:+UseContainerSupport \
           -XX:TieredStopAtLevel=1"
```

몇 가지 변경 의도를 설명드리면, Xmx를 320m 정도로 고정한 건 컨테이너 512m 한도에서 메타스페이스(128M)+코드캐시(48m)+스레드 스택+다이렉트 버퍼 등 힙 외 영역을 합쳐도 512m 안에 안전하게 들어오게 역산한 값입니다. 트래픽 늘어나면서 GC 로그 보고 조정하시면 됩니다.

UseSerialGC로 바꾼 이유는, G1GC는 힙이 수GB 단위로 커야 이점이 사는 GC라서, 320m처럼 작은 힙에서는 G1의 리전 관리 오버헤드가 오히려 손해입니다. 힙이 작고 단일 코어/저자원 환경이면 Serial GC가 오버헤드도 작고 메모리도 더 적게 씁니다. (트래픽이 커지고 GC pause가 문제되면 그때 G1이나 다른 GC로 다시 바꿔도 됩니다.)

TieredStopAtLevel=1은 JIT의 상위 컴파일 단계(C2)를 끄는 옵션인데, 이게 코드 캐시 사용량과 컴파일러 스레드가 쓰는 메모리를 줄여줍니다. 처리량이 아주 높은 서비스가 아니면 체감 성능 차이는 크지 않고 메모리/CPU 절약 효과가 있습니다.

ReservedCodeCacheSize도 기본값이 꽤 큰데, 작은 서비스에서는 48m 정도로 충분한 경우가 많습니다.

여기까지가 JVM 옵션 차원이고, 추가로 application.yml 쪽도 같이 봐야 할 부분이 있습니다. Spring Boot의 Tomcat 기본 스레드 풀이 max 200인데, 스레드 하나당 기본 스택이 1MB라서 풀이 다 차면 그것만으로 200MB까지 잠재적으로 먹을 수 있습니다(물론 lazy하게 commit되긴 하지만). 지금 인스턴스 규모면 이렇게 줄이는 게 안전합니다.

```yaml
server:
  tomcat:
    threads:
      max: 50
      min-spare: 5

spring:
  datasource:
    hikari:
      maximum-pool-size: 5
```

HikariCP 커넥션 풀도 기본값이 10인데, MySQL 컨테이너도 같은 인스턴스에서 446MiB를 쓰고 있는 걸 보면 커넥션 수를 무리하게 늘릴 여유가 없으니 5 정도로 시작하는 게 맞을 것 같습니다.

정리하면 우선순위는 1) deploy.sh에 `-m` 명시해서 JVM이 올바른 기준으로 퍼센트 계산하게 하기, 2) 퍼센트 대신 고정 Xms/Xmx로 전환, 3) GC를 Serial로, JIT 티어를 낮춰서 힙 외 오버헤드 줄이기, 4) Tomcat 스레드/Hikari 풀 사이즈 축소 순입니다. 이렇게 하면 빈 코드 기준으로 200~250MiB 선까지는 충분히 내려갈 거라고 봅니다.

## 분리한다면?
Dockerfile의 ENV JAVA_OPTS는 컨테이너 내부에서 java 프로세스가 실행될 때 적용되는 JVM 옵션입니다. 즉 "이 프로세스가 자기한테 주어진 자원을 어떻게 쓸지"를 정하는 겁니다. 반면 `-m`(메모리 limit)은 컨테이너 자체에 cgroup으로 거는 외부 제약인데, 이건 이미지 빌드 시점이 아니라 컨테이너를 실행하는 시점(`docker run`, `docker-compose`, k8s 등)에 결정되는 호스트/오케스트레이션 레벨의 설정입니다. Dockerfile 안에는 이걸 명시할 방법이 없습니다. 이미지는 "어떻게 실행될지"는 모르고 "실행되면 뭘 할지"만 담고 있는 거라서요.

그래서 `-XX:+UseContainerSupport`가 동작하는 방식을 다시 보면, JVM이 부팅할 때 자기가 속한 cgroup의 memory.max(또는 memory.limit_in_bytes)를 읽어서 "내가 쓸 수 있는 RAM이 이거구나"라고 판단합니다. `-m`을 안 주면 cgroup 자체에 제한이 안 걸려 있으니 JVM은 호스트 전체 메모리를 자기 한도로 인식하게 됩니다. 즉 Dockerfile의 퍼센트 옵션(`MaxRAMPercentage` 등)이 정확하게 동작하려면, 그 퍼센트를 계산할 기준값(limit)을 런타임에서 먼저 줘야 하는 거죠. 둘이 같은 목적을 위해 협력하는 두 레이어라서, 하나만 걸면 의미가 반쪽만 작동합니다.

docker-compose 분리는 결과적으로는 좋은 방향이라고 봅니다. 다만 blue-green 배포 구조 특성상 단순하게 옮기기엔 한 가지 함정이 있어서 짚어드릴게요.

compose v3 스펙에서 `deploy.resources.limits.memory`는 docker swarm 모드(`docker stack deploy`)에서만 실제로 적용됩니다. 일반 `docker compose up`으로 띄우는 단일 호스트 환경에서는 이 키가 무시됩니다. 대신 compose 서비스에 직접 `mem_limit`, `mem_swap_limit`을 써주면(이건 swarm 여부와 무관하게 compose CLI가 직접 docker run에 반영하는 옵션) 지금 환경에서 의도대로 동작합니다.

```yaml
services:
  mysql:
    image: mysql:8
    mem_limit: 512m
    mem_swap_limit: 512m
    restart: unless-stopped
    networks: [app-tier]

  nginx:
    image: nginx
    mem_limit: 64m
    restart: unless-stopped
    networks: [app-tier]
```

mysql, nginx, dbeaver처럼 단일 인스턴스로 고정되어 있는 서비스들은 compose로 옮기는 게 명확히 이득입니다. 네트워크, 재시작 정책, 메모리 제한을 한 파일에서 선언적으로 관리할 수 있고, 지금처럼 docker run 명령어를 여기저기 흩어놓고 운영하는 것보다 리뷰하기도, 재현하기도 훨씬 쉬워집니다.

문제는 swyp-spring-blue/green입니다. deploy.sh가 하는 핵심 로직, 즉 "현재 활성 프로파일 확인 → 반대쪽에 새 컨테이너 띄움 → 헬스체크 → nginx 전환 → 구버전 제거" 흐름은 compose의 기본 모델(서비스 하나당 컨테이너 하나, 이미지 바뀌면 recreate)과는 잘 안 맞습니다. compose로 이걸 표현하려면 서비스를 app-blue, app-green 두 개로 정의해두고, deploy.sh가 `docker compose up -d --no-deps app-green` 같은 식으로 compose 명령을 호출하는 식으로 바꿔야 합니다.

```yaml
services:
  app-blue:
    image: app5swyp6team/swyp-spring:latest
    container_name: swyp-spring-blue
    mem_limit: 512m
    mem_swap_limit: 512m
    ports: ["8081:8080"]
    environment:
      - SPRING_PROFILES_ACTIVE=prod
    networks: [app-tier]
    profiles: ["blue"]

  app-green:
    image: app5swyp6team/swyp-spring:latest
    container_name: swyp-spring-green
    mem_limit: 512m
    mem_swap_limit: 512m
    ports: ["8082:8080"]
    environment:
      - SPRING_PROFILES_ACTIVE=prod
    networks: [app-tier]
    profiles: ["green"]
```

이렇게 하면 deploy.sh 안의 `docker run ...` 블록을 `docker compose --profile ${TARGET} up -d ${TARGET_CONTAINER}`로, `docker rm -f`는 `docker compose stop/rm`으로 바꾸는 정도의 수정이면 됩니다. 메모리 limit이나 네트워크 설정은 compose 파일에 한 번만 박아두면 되니 deploy.sh가 더 간결해지는 효과도 있고요.

다만 솔직히 말씀드리면, 지금 인프라 규모(1.9GiB 단일 인스턴스, 컨테이너 4개)에서는 이 전환이 "당장 필수"는 아닙니다. 우선순위로 보면, 1) deploy.sh의 docker run에 `-m`만 먼저 추가해서 메모리 문제를 즉시 해결하고, 2) 그 다음 여유 있을 때 mysql/nginx/dbeaver를 compose로 옮겨서 관리 편의성을 높이고, 3) blue-green 앱 부분은 가장 마지막에, 필요하다고 느껴질 때 compose 기반으로 리팩토링하는 순서가 리스크 대비 효율이 좋을 것 같습니다. 지금 당장 셋 다 한 번에 바꾸면 디버깅 범위가 넓어져서 오히려 장애 원인 추적이 어려워질 수 있어서요.
