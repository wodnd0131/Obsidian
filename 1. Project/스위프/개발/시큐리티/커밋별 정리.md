public CustomClaims generateToken(String type, String accountId, String[] role) {
    // 1. 처음부터 Instant로 현재 시간 가져오기
    Instant now = Instant.now(); 
    
    // 2. Duration을 이용해 직관적으로 만료 시간 계산하기
    long expMillis = type.equals("ACCESS") ? ACCESS_EXP : REFRESH_EXP;
    Instant expiry = now.plusMillis(expMillis);

    // 3. Jwts.builder()에 Date 대신 Instant를 그대로 주입 (최신 jjwt 라이브러리 지원)
    String generatedToken = Jwts.builder()
                                    .subject(accountId)
                                    .claim("roles", role)
                                    .issuedAt(Date.from(now))       // 만약 구버전 jjwt라면 Date.from() 사용
                                    .expiration(Date.from(expiry)) // 최신 버전이면 .issuedAt(now) 처럼 Instant 바로 가능
                                    .signWith(key)
                                    .compact();

    // 4. 변환 과정(.toInstant()) 없이 깔끔하게 그대로 반환
    return new CustomClaims(accountId, generatedToken, now, expiry);
}
